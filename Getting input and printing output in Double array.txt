#include<stdio.h>
int main()
{
    int a[50][50],n,m;
    printf("Enter Column:");
    scanf("%d",&m);
    printf("Enter Row:");
    scanf("%d",&n);
    int i,j;
    for(i=0;i<m;i++)
    {
        for(j=0;j<n;j++){
        scanf("%d",&a[i][j]);
    } 
    }
    for(i=0;i<m;i++)
    {
        for(j=0;j<n;j++){
     printf("%d ",a[i][j]);
        }
        printf("\n");
    }
return 0;    
}
